1) SELECT CustomerID AS ID, CustomerName AS Customer
   FROM Customers;

2) SELECT CategoryName AS Name , Description AS Info 
   FROM Categories;

3) SELECT FirstName AS Name 
   FROM Employees;

4) SELECT SUM(Quantity) AS TotalQTY 
   FROM OrderDetails;

5) SELECT MAX(Price) AS MaxPrice , MIN(Price) AS MINPrice 
   FROM Products;
